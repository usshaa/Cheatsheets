‚Ä¢	FROM python:3.8-slim
‚Ä¢	
‚Ä¢	# Set the working directory
‚Ä¢	WORKDIR /app
‚Ä¢	
‚Ä¢	# Copy the local application to the container
‚Ä¢	COPY . /app
‚Ä¢	
‚Ä¢	# Install required packages
‚Ä¢	RUN pip install -r requirements.txt
‚Ä¢	
‚Ä¢	# Expose the port
‚Ä¢	EXPOSE 5000
‚Ä¢	
‚Ä¢	# Run the app
‚Ä¢	CMD ["python", "app.py"]

‚Ä¢	Build a Docker image: 
‚Ä¢	docker build -t my-image .

‚Ä¢	Run a container from the built image: 
‚Ä¢	docker run -p 5000:5000 my-image

üîπ 10. Docker Security Best Practices
‚Ä¢	Use official images: Always use official or verified images from Docker Hub or a trusted registry.
‚Ä¢	Avoid running containers as root: Use the USER directive in Dockerfile to specify a non-root user.
‚Ä¢	Limit resources: Use Docker resource flags to limit CPU and memory usage, e.g., --memory="500m".
